  public static boolean battle(Character player, int att, int hea)
  { 
  int health = player.getHealth();
  int opphth = hea;
  int i = 1;
  while(player.getHealth() > 0 && opphth > 0)
  {

  opphth = opphth - player.getAttack();
  System.out.println("\nRound " + i);
  System.out.println("Opponent health: " + opphth);
  health = health - att;
  System.out.println("Your health: " + health);
  i++;
  }
  if(health > 0 && health > opphth)
  {
  System.out.println("You win");
  return true;
  }
  else 
  {
  System.out.println("You lose, Game over");
  return false;
  }  }

Character Code public class Character { private String name; private int attack; private int health; private String type; private int level; private int numBattles;

public Character(String n, int att, int def, String c, int lvl, int b) { name = n; attack = att; health = def; level = lvl; numBattles = b; }

public String getName() { return name; }

public int getAttack() { return attack; }

public int getHealth() { return health; }

public String getType() { return type; }

public int getLvl() { return level; }

public int getbattles() { return numBattles; }

public void setName(String n) { name = n; }

public void setAttack(int att) { attack = att; }

public void setHealth(int hea) { health = hea; }

public void setClass(String t) { type = t; }

public void setlvl(int lvl) { level = lvl; }

public void upBattles() { numBattles += 1; }

}
